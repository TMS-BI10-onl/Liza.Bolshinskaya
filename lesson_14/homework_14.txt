CREATE VIEW BestSalesPersons AS
SELECT DISTINCT TOP 50 WITH TIES FirstName, LastName,
CASE
	WHEN cnt >=10 THEN 3
	WHEN cnt BETWEEN 5 AND 9 THEN 2
	WHEN cnt <5 THEN 1
END AS Rating
FROM (
	SELECT FirstName, LastName, OrderDate, COUNT(1) OVER (PARTITION BY OrderDate) AS cnt
	FROM [dbo].[SalesPerson] s JOIN [dbo].[Orders] o ON s.SalesPersonID=o.SalesPersonID
	GROUP BY FirstName, LastName, OrderDate
) T
ORDER BY FirstName, LastName DESC
